## 1. 運算式的三種表示法

| 表示法 | 格式 | 範例 (1+2)×3 |
|--------|------|--------------|
| **中置式 (Infix)** | 運算元 ⊕ 運算子 ⊕ 運算元 | `(1+2)*3` |
| **前置式 (Prefix)** | 運算子 ⊕ 運算元 ⊕ 運算元 | `*+123` |
| **後置式 (Postfix)** | 運算元 ⊕ 運算元 ⊕ 運算子 | `12+3*` |

## 2. 中置式轉後置式 (Shunting Yard)

```python
def infix_to_postfix(expression):
    """中置式轉後置式（Shunting Yard 演算法）"""
    precedence = {'+': 1, '-': 1, '*': 2, '/': 2, '%': 2}
    output = []
    stack = []

    for token in expression:
        if token.isdigit():  # 運算元
            output.append(token)
        elif token == '(':
            stack.append(token)
        elif token == ')':
            while stack and stack[-1] != '(':
                output.append(stack.pop())
            stack.pop()  # 移除 '('
        else:  # 運算子
            while (stack and stack[-1] != '(' and
                   precedence.get(stack[-1], 0) >= precedence.get(token, 0)):
                output.append(stack.pop())
            stack.append(token)

    while stack:
        output.append(stack.pop())

    return ''.join(output)

print(infix_to_postfix("1+2*3"))     # "123*+"
print(infix_to_postfix("(1+2)*3"))   # "12+3*"
```

## 3. 中置式轉前置式

轉換步驟：反轉字串 → 轉後置式（注意括號方向） → 再反轉：

```python
def infix_to_prefix(expression):
    """中置式轉前置式"""
    precedence = {'+': 1, '-': 1, '*': 2, '/': 2, '%': 2}

    # 1. 反轉字串，並交換 '(' 和 ')'
    reversed_expr = []
    for ch in reversed(expression):
        if ch == '(':
            reversed_expr.append(')')
        elif ch == ')':
            reversed_expr.append('(')
        else:
            reversed_expr.append(ch)

    # 2. 對反轉後的字串執行修改版的 Shunting Yard
    output = []
    stack = []
    for token in reversed_expr:
        if token.isdigit() or token.isalpha():
            output.append(token)
        elif token == '(':
            stack.append(token)
        elif token == ')':
            while stack and stack[-1] != '(':
                output.append(stack.pop())
            stack.pop()
        else:  # 運算子（注意：優先級順序要相反）
            while (stack and stack[-1] != '(' and
                   precedence.get(stack[-1], 0) > precedence.get(token, 0)):
                output.append(stack.pop())
            stack.append(token)

    while stack:
        output.append(stack.pop())

    # 3. 再反轉得到前置式
    return ''.join(reversed(output))

print(infix_to_prefix("1+2*3"))     # "+1*23"
print(infix_to_prefix("(1+2)*3"))   # "*+123"
```

## 4. 前置式轉後置式

假設運算元為單一字元（A~Z），運算子為 +-\*/：

```python
def prefix_to_postfix(expr):
    """前置式轉後置式（使用堆疊）"""
    stack = []
    # 從右到左掃描
    for ch in reversed(expr):
        if ch.isalpha():  # 運算元
            stack.append(ch)
        else:  # 運算子
            op1 = stack.pop()
            op2 = stack.pop()
            stack.append(op1 + op2 + ch)
    return stack[0]

print(prefix_to_postfix("*+ABC"))       # "AB+C*"
print(prefix_to_postfix("*-A/BC-/DKL")) # "ABC/-AK/L-*"
```

## 5. 後置式求值

```python
def evaluate_postfix(expr):
    """計算後置式的值（運算元為個位數整數）"""
    stack = []
    for ch in expr:
        if ch.isdigit():
            stack.append(int(ch))
        else:
            b = stack.pop()
            a = stack.pop()
            if ch == '+':
                stack.append(a + b)
            elif ch == '-':
                stack.append(a - b)
            elif ch == '*':
                stack.append(a * b)
            elif ch == '/':
                stack.append(a // b)
    return stack[0]

print(evaluate_postfix("23+5*"))  # (2+3)*5 = 25
```

## 6. 以最少移除來得到有效的括弧

給定一個包含小括號 `(` 和 `)` 的字串，刪除最少數量的括號使其變成合法的括號表示式：

```python
def remove_invalid_parentheses(s):
    """刪除最少括號得到合法表示式 — 回傳所有可能結果"""
    result = []

    def is_valid(t):
        """檢查括號是否合法"""
        count = 0
        for ch in t:
            if ch == '(':
                count += 1
            elif ch == ')':
                count -= 1
                if count < 0:
                    return False
        return count == 0

    def dfs(start, left_rem, right_rem, current):
        if left_rem == 0 and right_rem == 0:
            if is_valid(current):
                result.append(current)
            return

        for i in range(start, len(s)):
            # 去重：連續相同的括號只處理第一個
            if i > start and s[i] == s[i - 1]:
                continue

            ch = s[i]
            if ch == '(' and left_rem > 0:
                dfs(i + 1, left_rem - 1, right_rem, current + s[start:i])
            elif ch == ')' and right_rem > 0:
                dfs(i + 1, left_rem, right_rem - 1, current + s[start:i])

        # 加上剩餘部分
        dfs(len(s), left_rem, right_rem, current + s[start:])

    # 計算需要移除的左括號和右括號數量
    left_rem = right_rem = 0
    for ch in s:
        if ch == '(':
            left_rem += 1
        elif ch == ')':
            if left_rem > 0:
                left_rem -= 1
            else:
                right_rem += 1

    dfs(0, left_rem, right_rem, "")
    return result if result else [""]

print(remove_invalid_parentheses("()())()"))
# ['()()()', '(())()']
print(remove_invalid_parentheses(")("))
# ['']
```

## 7. 列出所有子字串

```python
def all_substrings(s):
    """列出字串的所有子字串（連續）"""
    result = []
    n = len(s)
    for i in range(n):
        for j in range(i + 1, n + 1):
            result.append(s[i:j])
    return result

print(all_substrings("abc"))
# ['a', 'ab', 'abc', 'b', 'bc', 'c']
```

## 8. 二維資料排序

```python
def sort_2d(data, key_col=0, reverse=False):
    """二維資料排序：根據某一欄排序"""
    return sorted(data, key=lambda row: row[key_col], reverse=reverse)

data = [
    (3, "Charlie"),
    (1, "Alice"),
    (2, "Bob")
]
print(sort_2d(data, key_col=0))       # [(1,'Alice'), (2,'Bob'), (3,'Charlie')]
print(sort_2d(data, key_col=1))       # [(1,'Alice'), (2,'Bob'), (3,'Charlie')]
```

### 依字串長度排序

```python
def sort_by_length(strings, reverse=False):
    """依字串長度排序"""
    return sorted(strings, key=len, reverse=reverse)

words = ["python", "java", "c", "javascript", "go"]
print(sort_by_length(words))
# ['c', 'go', 'java', 'python', 'javascript']
print(sort_by_length(words, reverse=True))
# ['javascript', 'python', 'java', 'go', 'c']
```

## 9. 將數值插入排序序列（二分搜尋）

```python
import bisect

def insert_into_sorted(arr, val):
    """將數值插入已排序序列，回傳應插入的位置"""
    pos = bisect.bisect_left(arr, val)
    return pos

arr = [1, 3, 5, 7, 9]
pos = insert_into_sorted(arr, 6)
print(f"6 應插入位置：{pos}")  # 3
arr.insert(pos, 6)
print(arr)  # [1, 3, 5, 6, 7, 9]

# 如果只想找位置而不插入
def find_insert_position(arr, val):
    """使用 bisect 找出數值應插入的位置"""
    # bisect_left：如果有相同值，插入到左邊
    # bisect_right：如果有相同值，插入到右邊
    return bisect.bisect_left(arr, val), bisect.bisect_right(arr, val)

arr = [1, 3, 3, 3, 5]
left, right = find_insert_position(arr, 3)
print(f"左插：{left}，右插：{right}")  # 左插：1，右插：4
```

## 時間複雜度

| 操作 | 時間 | 空間 |
|------|------|------|
| Shunting Yard (中→後) | O(n) | O(n) |
| 中→前 | O(n) | O(n) |
| 前→後 | O(n) | O(n) |
| 後置式求值 | O(n) | O(n) |
| 移除無效括號 | O(2ⁿ) 最差 | O(n) |
| 二維排序 | O(n log n) | O(n) |
