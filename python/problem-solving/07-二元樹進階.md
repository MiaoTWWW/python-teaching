## 1. 判斷是否為二元搜尋樹 (BST)

```python
def is_valid_bst(root):
    """判斷一棵樹是否為有效的二元搜尋樹"""
    def validate(node, low, high):
        if not node:
            return True
        if not (low < node.val < high):
            return False
        return (validate(node.left, low, node.val) and
                validate(node.right, node.val, high))
    return validate(root, float('-inf'), float('inf'))
```

## 2. 找出最近共同祖先 (LCA)

在二元樹中找到兩個節點的最近共同祖先：

```python
def lowest_common_ancestor(root, p, q):
    """二元樹的最近共同祖先"""
    if not root or root == p or root == q:
        return root

    left = lowest_common_ancestor(root.left, p, q)
    right = lowest_common_ancestor(root.right, p, q)

    if left and right:
        return root  # p 和 q 分別在左右子樹
    return left or right
```

### BST 版本的 LCA（更有效率）：

```python
def lca_bst(root, p, q):
    """二元搜尋樹的 LCA — O(h)"""
    while root:
        if p.val < root.val and q.val < root.val:
            root = root.left
        elif p.val > root.val and q.val > root.val:
            root = root.right
        else:
            return root
    return None
```

## 3. 二元樹的直徑 (Diameter)

直徑 = 任意兩個節點之間最長且不重複經過節點的邊數：

```python
def diameter_of_binary_tree(root):
    """二元樹的直徑"""
    diameter = [0]

    def height(node):
        if not node:
            return 0
        left_height = height(node.left)
        right_height = height(node.right)
        # 更新直徑：經過當前節點的最長路徑
        diameter[0] = max(diameter[0], left_height + right_height)
        return max(left_height, right_height) + 1

    height(root)
    return diameter[0]
```

## 4. 找出最大路徑總和 (Max Path Sum)

從任意節點到任意節點，求最大總和（可穿越父子路徑）：

```python
def max_path_sum(root):
    """二元樹中任意兩節點間的最大路徑和"""
    max_sum = [float('-inf')]

    def dfs(node):
        if not node:
            return 0

        # 左右子樹能提供的最大貢獻（如果為負則捨棄）
        left = max(0, dfs(node.left))
        right = max(0, dfs(node.right))

        # 以當前節點為轉折點的路徑和
        current_sum = node.val + left + right
        max_sum[0] = max(max_sum[0], current_sum)

        # 回傳包含當前節點的最大貢獻給父節點
        return node.val + max(left, right)

    dfs(root)
    return max_sum[0]
```

## 5. 水平反轉二元樹 (Invert Tree)

```python
def invert_tree(root):
    """水平反轉二元樹"""
    if not root:
        return None
    root.left, root.right = root.right, root.left
    invert_tree(root.left)
    invert_tree(root.right)
    return root
```

## 6. 比較兩棵二元樹是否相同

```python
def is_same_tree(p, q):
    """判斷兩棵二元樹是否完全相同"""
    if not p and not q:
        return True
    if not p or not q:
        return False
    return (p.val == q.val and
            is_same_tree(p.left, q.left) and
            is_same_tree(p.right, q.right))
```

## 7. 合併兩棵二元樹

```python
def merge_trees(t1, t2):
    """合併兩棵二元樹（對應節點值相加）"""
    if not t1 and not t2:
        return None
    if not t1:
        return t2
    if not t2:
        return t1

    merged = TreeNode(t1.val + t2.val)
    merged.left = merge_trees(t1.left, t2.left)
    merged.right = merge_trees(t1.right, t2.right)
    return merged
```

## 8. 從 root 到所有樹葉的路徑

```python
def binary_tree_paths(root):
    """輸出從 root 到所有樹葉的路徑"""
    def dfs(node, path, result):
        if not node:
            return
        path.append(str(node.val))
        if not node.left and not node.right:
            result.append("->".join(path))
        else:
            dfs(node.left, path, result)
            dfs(node.right, path, result)
        path.pop()

    result = []
    dfs(root, [], result)
    return result
```

## 9. 由 preorder 及 inorder 建立二元樹

```python
def build_tree(preorder, inorder):
    """從前序和中序走訪重建二元樹"""
    if not preorder or not inorder:
        return None

    root_val = preorder[0]
    root = TreeNode(root_val)

    # 在中序中找到 root 的位置
    root_idx = inorder.index(root_val)

    # 左子樹：preorder[1:root_idx+1], inorder[:root_idx]
    root.left = build_tree(preorder[1:root_idx + 1], inorder[:root_idx])
    # 右子樹：preorder[root_idx+1:], inorder[root_idx+1:]
    root.right = build_tree(preorder[root_idx + 1:], inorder[root_idx + 1:])

    return root

# 也可以用 Hash Map 最佳化
def build_tree_optimized(preorder, inorder):
    in_map = {val: i for i, val in enumerate(inorder)}

    def build(pre_start, pre_end, in_start, in_end):
        if pre_start > pre_end or in_start > in_end:
            return None

        root_val = preorder[pre_start]
        root = TreeNode(root_val)
        root_idx = in_map[root_val]
        left_size = root_idx - in_start

        root.left = build(pre_start + 1, pre_start + left_size,
                          in_start, root_idx - 1)
        root.right = build(pre_start + left_size + 1, pre_end,
                           root_idx + 1, in_end)
        return root

    return build(0, len(preorder) - 1, 0, len(inorder) - 1)

preorder = [3, 9, 20, 15, 7]
inorder = [9, 3, 15, 20, 7]
root = build_tree(preorder, inorder)
#     3
#    / \
#   9  20
#      / \
#     15  7
```

## 10. 判斷是否為 AVL 樹

AVL 樹：是二元搜尋樹，且每個節點的左右子樹高度差 ≤ 1：

```python
def is_avl(root):
    """判斷一棵樹是否為 AVL 樹"""
    def check(node):
        if not node:
            return 0  # 高度

        left = check(node.left)
        if left == -1:
            return -1  # 不平衡

        right = check(node.right)
        if right == -1:
            return -1

        if abs(left - right) > 1:
            return -1

        return max(left, right) + 1

    return check(root) != -1
```

## 11. 計算各節點的左右子樹高度

```python
def node_heights(root):
    """計算每個節點的左右子樹高度"""
    heights = {}

    def dfs(node):
        if not node:
            return 0
        left = dfs(node.left)
        right = dfs(node.right)
        heights[node.val] = (left, right)
        return max(left, right) + 1

    dfs(root)
    return heights
```

## 12. 刪除特定值的樹葉節點

若刪除導致新葉節點也符合條件，繼續刪除：

```python
def remove_leaf_nodes(root, target):
    """刪除所有值等於 target 的樹葉節點（連鎖刪除）"""
    if not root:
        return None

    root.left = remove_leaf_nodes(root.left, target)
    root.right = remove_leaf_nodes(root.right, target)

    # 如果當前節點已變成 target 值的葉節點，刪除它
    if not root.left and not root.right and root.val == target:
        return None

    return root
```

## 13. N 元樹 (General Tree)

分支度為 5 的樹（每個節點最多有 5 個子節點）：

```python
class NaryTreeNode:
    def __init__(self, val=0, children=None):
        self.val = val
        self.children = children if children else []

def create_nary_tree(adj_list, root_val=1):
    """從 adjacency list 建立 N 元樹"""
    from collections import deque

    if root_val not in adj_list:
        return NaryTreeNode(root_val)

    root = NaryTreeNode(root_val)
    queue = deque([root])

    while queue:
        node = queue.popleft()
        if node.val in adj_list:
            for child_val in adj_list[node.val]:
                child = NaryTreeNode(child_val)
                node.children.append(child)
                queue.append(child)

    return root

def bfs_nary(root):
    """N 元樹的 BFS 走訪"""
    if not root:
        return
    queue = deque([root])
    while queue:
        node = queue.popleft()
        print(node.val, end=" ")
        for child in node.children:
            queue.append(child)

def dfs_nary(root):
    """N 元樹的 DFS 前序走訪"""
    if not root:
        return
    print(root.val, end=" ")
    for child in root.children:
        dfs_nary(child)

# 範例
adj = {
    1: [2, 3, 4],
    2: [5, 6],
    4: [7]
}
nary_root = create_nary_tree(adj, 1)
print("BFS: ", end="")
bfs_nary(nary_root)       # 1 2 3 4 5 6 7
print("\nDFS: ", end="")
dfs_nary(nary_root)       # 1 2 5 6 3 4 7
```

## 二元樹常見操作速查

| 操作 | 方法 | 時間 |
|------|------|------|
| 判斷 BST | `is_valid_bst` | O(n) |
| LCA | 遞迴查找 | O(n) |
| 直徑 | 深度優先 + 全域變數 | O(n) |
| 最大路徑和 | DFS + 貢獻值 | O(n) |
| 反轉樹 | 左右交換遞迴 | O(n) |
| 判斷相同 | 同步走訪 | O(n) |
| 合併樹 | 同步走訪相加 | O(n) |
| 重建樹 (pre+in) | 遞迴分割 | O(n) |
| AVL 判斷 | 後序檢查 | O(n) |
