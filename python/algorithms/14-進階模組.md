## collections

```python
from collections import Counter, defaultdict, deque, OrderedDict

# Counter — 計數器
items = ['a', 'b', 'a', 'c', 'b', 'a']
count = Counter(items)
print(count)                # Counter({'a': 3, 'b': 2, 'c': 1})
print(count.most_common(2)) # [('a', 3), ('b', 2)]
print(count['d'])           # 0（不存在回傳 0，不會拋錯）

# defaultdict — 預設值字典
d = defaultdict(list)       # 預設為空串列
d['students'].append('Alice')
d['students'].append('Bob')
print(d)                    # defaultdict(<class 'list'>, {'students': ['Alice', 'Bob']})

# defaultdict(int) 適合計數
freq = defaultdict(int)
for ch in "hello":
    freq[ch] += 1
print(dict(freq))           # {'h': 1, 'e': 1, 'l': 2, 'o': 1}

# deque — 雙端佇列
dq = deque([1, 2, 3])
dq.append(4)                # deque([1, 2, 3, 4])
dq.appendleft(0)            # deque([0, 1, 2, 3, 4])
dq.pop()                    # 4
dq.popleft()                # 0
print(dq)                   # deque([1, 2, 3])
dq.rotate(1)                # 右旋轉 → deque([3, 1, 2])
dq.rotate(-1)               # 左旋轉 → deque([1, 2, 3])
```

## itertools

```python
import itertools

# count — 無限計數器
for i, val in zip(range(5), itertools.count(10, 2)):
    print(val, end=" ")     # 10 12 14 16 18

# cycle — 無限循環
colors = itertools.cycle(['紅', '黃', '藍'])
for _ in range(6):
    print(next(colors), end=" ")  # 紅 黃 藍 紅 黃 藍

# chain — 串接多個可迭代物件
combined = itertools.chain([1, 2], [3, 4], [5])
print(list(combined))       # [1, 2, 3, 4, 5]

# groupby — 分組
data = [("A", 1), ("A", 2), ("B", 3), ("B", 4)]
for key, group in itertools.groupby(data, key=lambda x: x[0]):
    print(key, list(group))
# A [('A', 1), ('A', 2)]
# B [('B', 3), ('B', 4)]

# product — 笛卡爾積（巢狀迴圈的替代）
for p in itertools.product([1, 2], ['a', 'b']):
    print(p)
# (1, 'a'), (1, 'b'), (2, 'a'), (2, 'b')

# permutations / combinations
print(list(itertools.permutations([1, 2, 3], 2)))
# [(1,2), (1,3), (2,1), (2,3), (3,1), (3,2)]
print(list(itertools.combinations([1, 2, 3], 2)))
# [(1,2), (1,3), (2,3)]

# product 搭配 repeat — 多重迴圈
# 等同於巢狀 for i in range(2): for j in range(2):
for bits in itertools.product([0, 1], repeat=2):
    print(bits)
# (0,0), (0,1), (1,0), (1,1)
```

## string

```python
import string

print(string.ascii_lowercase)  # abcdefghijklmnopqrstuvwxyz
print(string.ascii_uppercase)  # ABCDEFGHIJKLMNOPQRSTUVWXYZ
print(string.digits)           # 0123456789
print(string.hexdigits)        # 0123456789abcdefABCDEF
print(string.punctuation)      # !"#$%&'()*+,-./:;<=>?@[\]^_`{|}~
print(string.whitespace)       # 空白字元（空白、tab、換行等）

# 實用範例：產生隨機密碼
import random
password = ''.join(random.choices(string.ascii_letters + string.digits, k=8))
print(password)  # 例如：aK3pQ9xZ
```

## statistics

```python
from statistics import mean, median, mode, stdev, variance

data = [1, 2, 2, 3, 4, 5, 5, 5, 6]

print(mean(data))      # 3.666...
print(median(data))    # 4
print(mode(data))      # 5（出現最多次）
print(stdev(data))     # 標準差
print(variance(data))  # 變異數
```

## decimal

```python
from decimal import Decimal, getcontext

# 解決浮點數精度問題
print(0.1 + 0.2)              # 0.30000000000000004
print(Decimal('0.1') + Decimal('0.2'))  # 0.3

# 設定精確度
getcontext().prec = 6
print(Decimal(1) / Decimal(7))  # 0.142857

# 金融計算
price = Decimal('19.99')
tax_rate = Decimal('0.05')
total = price + price * tax_rate
print(total)  # 20.9895（保留精確度）
```

## fractions

```python
from fractions import Fraction

# 建立分數
f1 = Fraction(1, 3)      # 1/3
f2 = Fraction(2, 6)      # 自動化簡為 1/3
print(f1 + f2)            # 2/3
print(f1 * 3)             # 1
print(Fraction('0.25'))   # 1/4

# 處理有理數精確計算
result = Fraction(1, 3) + Fraction(1, 6)
print(result)             # 1/2
print(float(result))      # 0.5
```

## datetime

```python
from datetime import datetime, date, timedelta

# 取得目前時間
now = datetime.now()
print(now)                # 2026-06-14 15:30:00

# 格式化
print(now.strftime("%Y/%m/%d %H:%M"))  # 2026/06/14 15:30
print(now.strftime("%A"))              # Sunday

# 字串轉 datetime
dt = datetime.strptime("2026-06-14", "%Y-%m-%d")
print(dt)                 # 2026-06-14 00:00:00

# 日期運算
today = date.today()
tomorrow = today + timedelta(days=1)
yesterday = today - timedelta(days=1)
print(tomorrow - yesterday)  # 2 days, 0:00:00

# 計算兩個日期間的天數
d1 = date(2026, 1, 1)
d2 = date(2026, 12, 31)
print((d2 - d1).days)  # 364
```
