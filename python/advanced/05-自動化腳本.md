### 批次重新命名

```python
from pathlib import Path
import re

def batch_rename(directory, pattern, replacement):
    """批次重新命名檔案"""
    path = Path(directory)

    for file in path.iterdir():
        if file.is_file():
            new_name = re.sub(pattern, replacement, file.name)
            if new_name != file.name:
                new_path = file.with_name(new_name)
                file.rename(new_path)
                print(f"重新命名: {file.name} → {new_name}")

# 使用：將所有檔案中的空格取代為底線
batch_rename("./documents", r"\s+", "_")

# 使用：將所有 .txt 改為 .md
def change_extension(directory, old_ext, new_ext):
    path = Path(directory)
    for file in path.glob(f"*{old_ext}"):
        new_path = file.with_suffix(new_ext)
        file.rename(new_path)
        print(f"變更副檔名: {file.name} → {new_path.name}")

change_extension("./docs", ".txt", ".md")
```

### 檔案整理器

```python
from pathlib import Path
import shutil
from datetime import datetime

class FileOrganizer:
    """自動整理下載目錄的檔案"""

    FILE_CATEGORIES = {
        "圖片": [".jpg", ".jpeg", ".png", ".gif", ".bmp", ".svg"],
        "文件": [".pdf", ".doc", ".docx", ".xlsx", ".pptx", ".txt"],
        "壓縮檔": [".zip", ".rar", ".7z", ".tar", ".gz"],
        "程式碼": [".py", ".js", ".html", ".css", ".java", ".cpp"],
        "影片": [".mp4", ".avi", ".mkv", ".mov"],
        "音樂": [".mp3", ".wav", ".flac", ".aac"],
    }

    def __init__(self, target_dir):
        self.target_dir = Path(target_dir)

    def organize(self):
        """整理目標目錄中的所有檔案"""
        if not self.target_dir.exists():
            print(f"目錄不存在: {self.target_dir}")
            return

        organized_count = 0

        for file in self.target_dir.iterdir():
            if file.is_file():
                category = self._get_category(file)
                if category:
                    self._move_to_category(file, category)
                    organized_count += 1

        print(f"已整理 {organized_count} 個檔案！")

    def _get_category(self, file):
        """根據副檔名分類"""
        for category, extensions in self.FILE_CATEGORIES.items():
            if file.suffix.lower() in extensions:
                return category
        return "其他"

    def _move_to_category(self, file, category):
        """將檔案移動到對應分類目錄"""
        dest_dir = self.target_dir / category
        dest_dir.mkdir(exist_ok=True)
        dest_path = dest_dir / file.name

        # 避免覆蓋
        if dest_path.exists():
            timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
            dest_path = dest_dir / f"{file.stem}_{timestamp}{file.suffix}"

        shutil.move(str(file), str(dest_path))
        print(f"移動: {file.name} → {category}/")

# 使用
organizer = FileOrganizer("C:/Users/YourName/Downloads")
organizer.organize()
```

## 系統監控腳本

```python
import psutil
import json
from datetime import datetime

def get_system_info():
    """取得系統資訊"""
    info = {
        "timestamp": datetime.now().isoformat(),
        "cpu": {
            "usage_percent": psutil.cpu_percent(interval=1),
            "count": psutil.cpu_count(),
            "frequency": psutil.cpu_freq()._asdict() if psutil.cpu_freq() else {}
        },
        "memory": {
            "total": psutil.virtual_memory().total,
            "available": psutil.virtual_memory().available,
            "usage_percent": psutil.virtual_memory().percent
        },
        "disk": {
            "total": psutil.disk_usage("/").total,
            "free": psutil.disk_usage("/").free,
            "usage_percent": psutil.disk_usage("/").percent
        },
        "network": {
            "bytes_sent": psutil.net_io_counters().bytes_sent,
            "bytes_recv": psutil.net_io_counters().bytes_recv
        }
    }
    return info

def monitor_system(interval=5, duration=60):
    """監控系統資源使用情況"""
    import time

    print(f"開始監控系統（間隔 {interval} 秒，總計 {duration} 秒）")
    start_time = time.time()

    while time.time() - start_time < duration:
        info = get_system_info()
        cpu = info["cpu"]["usage_percent"]
        mem = info["memory"]["usage_percent"]
        timestamp = info["timestamp"][11:19]

        print(f"[{timestamp}] CPU: {cpu:5.1f}% | 記憶體: {mem:5.1f}%")

        # 如果使用率過高，發出警告
        if cpu > 80:
            print(f"  ⚠️ CPU 使用率過高！")
        if mem > 90:
            print(f"  ⚠️ 記憶體即將耗盡！")

        time.sleep(interval)
```

## 備份腳本

```python
from pathlib import Path
import shutil
import json
from datetime import datetime

class BackupManager:
    """自動備份管理工具"""

    def __init__(self, source_dir, backup_dir, config_file="backup_config.json"):
        self.source = Path(source_dir)
        self.backup = Path(backup_dir)
        self.config_file = Path(config_file)
        self.config = self._load_config()

    def _load_config(self):
        """載入備份設定"""
        if self.config_file.exists():
            return json.loads(self.config_file.read_text(encoding="utf-8"))
        return {
            "exclude_extensions": [".tmp", ".log", ".cache"],
            "exclude_dirs": ["__pycache__", ".git", "node_modules"],
            "max_backups": 5
        }

    def backup(self):
        """執行備份"""
        if not self.source.exists():
            print(f"來源目錄不存在: {self.source}")
            return

        # 建立備份目錄（含時間戳）
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        backup_path = self.backup / f"backup_{timestamp}"
        backup_path.mkdir(parents=True, exist_ok=True)

        copied = 0
        skipped = 0

        for file in self.source.rglob("*"):
            if file.is_file():
                # 檢查是否需要排除
                if self._should_exclude(file):
                    skipped += 1
                    continue

                # 保持目錄結構
                relative_path = file.relative_to(self.source)
                dest_path = backup_path / relative_path
                dest_path.parent.mkdir(parents=True, exist_ok=True)

                shutil.copy2(file, dest_path)
                copied += 1

        print(f"備份完成！")
        print(f"  複製: {copied} 個檔案")
        print(f"  跳過: {skipped} 個檔案")
        print(f"  位置: {backup_path}")

        # 清理舊備份
        self._cleanup_old_backups()

    def _should_exclude(self, file):
        """檢查是否應該排除此檔案"""
        # 檢查副檔名
        if file.suffix.lower() in self.config["exclude_extensions"]:
            return True

        # 檢查目錄名稱
        for part in file.parts:
            if part in self.config["exclude_dirs"]:
                return True

        return False

    def _cleanup_old_backups(self):
        """刪除超過最大保留數量的舊備份"""
        if not self.backup.exists():
            return

        backups = sorted(self.backup.glob("backup_*"))
        max_backups = self.config.get("max_backups", 5)

        while len(backups) > max_backups:
            old_backup = backups.pop(0)
            shutil.rmtree(old_backup)
            print(f"刪除舊備份: {old_backup.name}")

# 使用
# manager = BackupManager(

#     backup_dir="./backups"
# )
# manager.backup()
```

## 批次圖片處理

```python
from pathlib import Path
from PIL import Image, ImageFilter

class ImageProcessor:
    """批次圖片處理工具"""

    SUPPORTED_FORMATS = {".jpg", ".jpeg", ".png", ".bmp"}

    def __init__(self, input_dir, output_dir=None):
        self.input_dir = Path(input_dir)
        self.output_dir = Path(output_dir) if output_dir else self.input_dir / "processed"
        self.output_dir.mkdir(exist_ok=True)

    def resize_all(self, max_width=800, max_height=600):
        """批次調整圖片大小"""
        for img_path in self._get_images():
            try:
                with Image.open(img_path) as img:
                    original_size = img.size
                    img.thumbnail((max_width, max_height), Image.LANCZOS)

                    output_path = self.output_dir / img_path.name
                    img.save(output_path)

                    print(f"調整: {img_path.name} "
                          f"({original_size[0]}x{original_size[1]} → "
                          f"{img.size[0]}x{img.size[1]})")
            except Exception as e:
                print(f"錯誤 {img_path.name}: {e}")

    def convert_format(self, target_format="PNG"):
        """批次轉換圖片格式"""
        for img_path in self._get_images():
            try:
                with Image.open(img_path) as img:
                    output_name = img_path.stem + f".{target_format.lower()}"
                    output_path = self.output_dir / output_name
                    img.save(output_path, target_format)

                    print(f"轉換: {img_path.name} → {output_name}")
            except Exception as e:
                print(f"錯誤 {img_path.name}: {e}")

    def apply_filter(self, filter_name="BLUR"):
        """批次套用濾鏡"""
        filters = {
            "BLUR": ImageFilter.BLUR,
            "CONTOUR": ImageFilter.CONTOUR,
            "DETAIL": ImageFilter.DETAIL,
            "EDGE_ENHANCE": ImageFilter.EDGE_ENHANCE,
            "SHARPEN": ImageFilter.SHARPEN,
            "SMOOTH": ImageFilter.SMOOTH,
        }

        selected_filter = filters.get(filter_name.upper())
        if not selected_filter:
            print(f"不支援的濾鏡: {filter_name}")
            return

        for img_path in self._get_images():
            try:
                with Image.open(img_path) as img:
                    filtered = img.filter(selected_filter)

                    output_path = self.output_dir / img_path.name
                    filtered.save(output_path)

                    print(f"套用 {filter_name}: {img_path.name}")
            except Exception as e:
                print(f"錯誤 {img_path.name}: {e}")

    def _get_images(self):
        """取得所有支援的圖片檔案"""
        images = []
        for ext in self.SUPPORTED_FORMATS:
            images.extend(self.input_dir.glob(f"*{ext}"))
            images.extend(self.input_dir.glob(f"*{ext.upper()}"))
        return images
```
