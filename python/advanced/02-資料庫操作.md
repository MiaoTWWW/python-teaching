## SQLite 基本操作

SQLite 是 Python 內建的輕量級資料庫，不需要額外安裝。

### 連線與建立資料表

```python
import sqlite3

# 連線資料庫（如果不存在會自動建立）
conn = sqlite3.connect("school.db")
cursor = conn.cursor()

# 建立資料表
cursor.execute("""
    CREATE TABLE IF NOT EXISTS students (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        name TEXT NOT NULL,
        age INTEGER,
        score REAL,
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    )
""")

# 儲存變更
conn.commit()
conn.close()
```

### 新增資料（CREATE）

```python
import sqlite3

conn = sqlite3.connect("school.db")
cursor = conn.cursor()

# 方式一：直接寫入 SQL
cursor.execute(
    "INSERT INTO students (name, age, score) VALUES (?, ?, ?)",
    ("Alice", 20, 95.5)
)

# 方式二：一次新增多筆
students_data = [
    ("Bob", 22, 88.0),
    ("Charlie", 21, 92.5),
    ("Diana", 23, 85.5)
]
cursor.executemany(
    "INSERT INTO students (name, age, score) VALUES (?, ?, ?)",
    students_data
)

conn.commit()
print(f"新增 {cursor.rowcount} 筆資料")
conn.close()
```

### 查詢資料（READ）

```python
import sqlite3

conn = sqlite3.connect("school.db")
# 設定 row_factory 讓查詢結果可以使用欄位名稱存取
conn.row_factory = sqlite3.Row
cursor = conn.cursor()

# 查詢所有資料
cursor.execute("SELECT * FROM students")
rows = cursor.fetchall()

for row in rows:
    print(f"ID: {row['id']}, 姓名: {row['name']}, "
          f"年齡: {row['age']}, 成績: {row['score']}")

# 條件查詢
cursor.execute(
    "SELECT * FROM students WHERE score >= ? ORDER BY score DESC",
    (90,)
)
top_students = cursor.fetchall()
print(f"\n成績 >= 90 的學生: {len(top_students)} 人")

# 聚合查詢
cursor.execute("""
    SELECT 
        AVG(score) as avg_score,
        MAX(score) as max_score,
        MIN(score) as min_score,
        COUNT(*) as total
    FROM students
""")
stats = cursor.fetchone()
print(f"\n平均成績: {stats['avg_score']:.1f}")
print(f"最高分: {stats['max_score']}")
print(f"最低分: {stats['min_score']}")
print(f"總人數: {stats['total']}")

conn.close()
```

### 更新與刪除資料

```python
import sqlite3

conn = sqlite3.connect("school.db")
cursor = conn.cursor()

# 更新資料
cursor.execute(
    "UPDATE students SET score = ? WHERE name = ?",
    (97.0, "Alice")
)
print(f"更新 {cursor.rowcount} 筆資料")

# 刪除資料
cursor.execute(
    "DELETE FROM students WHERE score < ?",
    (60,)
)
print(f"刪除 {cursor.rowcount} 筆資料")

conn.commit()
conn.close()
```

## SQLAlchemy ORM

ORM（Object-Relational Mapping）讓你用**物件導向的方式**操作資料庫。

### 安裝

```bash
pip install sqlalchemy
```

### 定義模型

```python
from sqlalchemy import create_engine, Column, Integer, String, Float, DateTime, func
from sqlalchemy.orm import declarative_base, sessionmaker
from datetime import datetime

# 建立資料庫連線
engine = create_engine("sqlite:///school.db", echo=True)
Base = declarative_base()

# 定義模型
class Student(Base):
    __tablename__ = "students_orm"

    id = Column(Integer, primary_key=True, autoincrement=True)
    name = Column(String, nullable=False)
    age = Column(Integer)
    score = Column(Float)
    created_at = Column(DateTime, default=datetime.now)

    def __repr__(self):
        return f"<Student(name='{self.name}', age={self.age}, score={self.score})>"

# 建立資料表
Base.metadata.create_all(engine)
```

### CRUD 操作

```python
from sqlalchemy.orm import sessionmaker

# 建立 Session
Session = sessionmaker(bind=engine)
session = Session()

# 新增
student1 = Student(name="Alice", age=20, score=95.5)
student2 = Student(name="Bob", age=22, score=88.0)
session.add(student1)
session.add(student2)
session.commit()

# 查詢
all_students = session.query(Student).all()
alice = session.query(Student).filter_by(name="Alice").first()
top_scores = session.query(Student).filter(Student.score >= 90).all()

# 排序
sorted_students = session.query(Student).order_by(Student.score.desc()).all()

# 聚合
from sqlalchemy import func
stats = session.query(
    func.avg(Student.score).label("avg_score"),
    func.max(Student.score).label("max_score"),
    func.count(Student.id).label("total")
).first()

print(f"平均成績: {stats.avg_score:.1f}")

# 更新
alice = session.query(Student).filter_by(name="Alice").first()
alice.score = 97.0
session.commit()

# 刪除
low_scores = session.query(Student).filter(Student.score < 60)
low_scores.delete()
session.commit()

session.close()
```

### 關聯資料表

```python
from sqlalchemy import ForeignKey
from sqlalchemy.orm import relationship

class Course(Base):
    __tablename__ = "courses"

    id = Column(Integer, primary_key=True, autoincrement=True)
    name = Column(String, nullable=False)
    teacher = Column(String)

class Enrollment(Base):
    __tablename__ = "enrollments"

    id = Column(Integer, primary_key=True, autoincrement=True)
    student_id = Column(Integer, ForeignKey("students_orm.id"))
    course_id = Column(Integer, ForeignKey("courses.id"))

    student = relationship("Student", backref="enrollments")
    course = relationship("Course", backref="enrollments")

# 建立資料表
Base.metadata.create_all(engine)

# 使用
session = Session()

# 新增課程
course = Course(name="Python 程式設計", teacher="Dr. Wang")
session.add(course)
session.commit()

# 學生選課
student = session.query(Student).filter_by(name="Alice").first()
enrollment = Enrollment(student_id=student.id, course_id=course.id)
session.add(enrollment)
session.commit()

# 查詢學生的選課
alice = session.query(Student).filter_by(name="Alice").first()
for enrollment in alice.enrollments:
    print(f"{alice.name} 選了 {enrollment.course.name}")

session.close()
```

## 資料庫設計基本原則

### 正規化

```python
# ❌ 不好的設計：全部塞在同一張表
students = [
    ("Alice", "Python", "Dr. Wang", 95),
    ("Alice", "Math", "Prof. Li", 88),
    ("Bob", "Python", "Dr. Wang", 92),
]

# ✅ 好的設計：拆分成多張表
# students: id, name, age
# courses: id, name, teacher
# enrollments: student_id, course_id, score
```

### 使用 Context Manager

```python
from contextlib import contextmanager

@contextmanager
def get_db():
    """資料庫連線上下文管理器"""
    conn = sqlite3.connect("school.db")
    try:
        yield conn
        conn.commit()
    except Exception:
        conn.rollback()
        raise
    finally:
        conn.close()

# 使用
with get_db() as conn:
    cursor = conn.cursor()
    cursor.execute("SELECT * FROM students")
    data = cursor.fetchall()
```

## 重點整理

| 操作 | SQLite（原生） | SQLAlchemy（ORM） |
|------|---------------|------------------|
| 連線 | `sqlite3.connect()` | `create_engine()` |
| 新增 | `INSERT INTO` | `session.add()` |
| 查詢 | `SELECT` | `session.query()` |
| 更新 | `UPDATE` | 修改物件屬性 |
| 刪除 | `DELETE` | `session.delete()` |
