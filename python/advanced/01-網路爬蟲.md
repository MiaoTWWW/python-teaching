## 什麼是網路爬蟲？

網路爬蟲（Web Crawler / Web Scraper）是一種**自動化擷取網頁資料**的程式。它可以幫助我們：

- 📥 收集資料集
- 📊 監控價格變化
- 🔍 搜尋引擎索引
- 📰 新聞聚合

## HTTP 基礎

### 請求和回應

```python
import requests

# GET 請求
response = requests.get("https://httpbin.org/get")
print(response.status_code)  # 200
print(response.text)         # HTML 內容

# 帶參數的 GET 請求
params = {"q": "python", "page": 1}
response = requests.get("https://httpbin.org/get", params=params)
print(response.url)  # 包含查詢參數的 URL

# POST 請求
data = {"username": "alice", "password": "123456"}
response = requests.post("https://httpbin.org/post", data=data)
print(response.json())
```

### 處理回應

```python
import requests

response = requests.get("https://api.github.com/users/python")

# 檢查狀態碼
if response.status_code == 200:
    print("成功！")
elif response.status_code == 404:
    print("資源不存在")
elif response.status_code == 403:
    print("禁止存取")

# 不同格式的回應
print(response.text)          # 原始文字
print(response.json())        # JSON 格式（自動解析）
print(response.content)       # 二進位內容（圖片等）

# 回應標頭
print(response.headers)
print(response.headers.get("Content-Type"))
```

```python
import requests

# 取得網頁內容
response = requests.get("https://example.com")
soup = BeautifulSoup(response.text, "html.parser")

# 基本搜尋
print(soup.title)                # <title>...</title>
print(soup.title.text)           # 標題文字
print(soup.find("h1"))          # 第一個 h1
print(soup.find_all("p"))       # 所有 p 標籤

# 使用 CSS 選擇器
print(soup.select(".main-content"))  # class 為 main-content
print(soup.select("#header"))        # id 為 header
print(soup.select("div > p"))        # div 底下的 p
```

### 實際爬蟲範例

```python
import requests
from bs4 import BeautifulSoup
import time

def fetch_news_titles(url):
    """擷取網頁中的新聞標題"""
    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36"
    }

    try:
        response = requests.get(url, headers=headers, timeout=10)
        response.raise_for_status()
        response.encoding = "utf-8"

        soup = BeautifulSoup(response.text, "html.parser")

        # 尋找標題（根據實際網頁結構調整）
        titles = soup.find_all("h2")
        for i, title in enumerate(titles[:10], 1):
            print(f"{i}. {title.text.strip()}")

    except requests.RequestException as e:
        print(f"請求失敗: {e}")

# 執行爬蟲
fetch_news_titles("https://example-news-site.com")
```

### 禮貌性延遲

```python
import time
import requests

def polite_crawl(urls, delay=1):
    """禮貌性爬蟲：每次請求之間加入延遲"""
    for url in urls:
        print(f"正在爬取: {url}")
        response = requests.get(url)
        # 處理回應...
        time.sleep(delay)  # 等待指定秒數

# 使用
urls = ["https://example.com/page1", "https://example.com/page2"]
polite_crawl(urls, delay=2)  # 每 2 秒請求一次
```

### 設定請求間隔

```python
class Crawler:
    def __init__(self, base_delay=1):
        self.base_delay = base_delay
        self.last_request_time = 0

    def fetch(self, url):
        """在請求之間確保有足夠的間隔"""
        elapsed = time.time() - self.last_request_time
        if elapsed < self.base_delay:
            time.sleep(self.base_delay - elapsed)

        response = requests.get(url)
        self.last_request_time = time.time()
        return response
```

## 完整實例：爬取 GitHub 熱門專案

```python
import requests
from bs4 import BeautifulSoup
import time
import json

class GitHubTrending:
    """GitHub 熱門專案爬蟲"""

    BASE_URL = "https://github.com/trending"

    def __init__(self, language="python", since="weekly"):
        self.language = language
        self.since = since
        self.headers = {
            "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36"
        }

    def fetch_trending(self):
        """取得熱門專案列表"""
        url = f"{self.BASE_URL}/{self.language}?since={self.since}"
        print(f"正在爬取: {url}")

        response = requests.get(url, headers=self.headers)
        response.raise_for_status()

        soup = BeautifulSoup(response.text, "html.parser")
        repositories = []

        for article in soup.select("article.Box-row"):
            repo = self._parse_repository(article)
            if repo:
                repositories.append(repo)

            time.sleep(0.1)  # 禮貌性延遲

        return repositories

    def _parse_repository(self, article):
        """解析單個專案資訊"""
        try:
            # 專案名稱
            name_elem = article.select_one("h2 a")
            name = name_elem.text.strip().replace(" ", "")

            # 描述
            desc_elem = article.select_one("p")
            description = desc_elem.text.strip() if desc_elem else ""

            # 星星數
            stars_elem = article.select_one(".d-inline-block.ml-3")
            stars = stars_elem.text.strip() if stars_elem else "0"

            # 今日星星
            today_elem = article.select_one(".float-sm-right")
            today_stars = today_elem.text.strip() if today_elem else ""

            return {
                "name": name,
                "url": f"https://github.com/{name}",
                "description": description,
                "stars": stars,
                "today_stars": today_stars
            }
        except Exception as e:
            print(f"解析錯誤: {e}")
            return None

# 使用
crawler = GitHubTrending(language="python", since="weekly")
repos = crawler.fetch_trending()

# 輸出結果
for i, repo in enumerate(repos[:10], 1):
    print(f"{i}. {repo['name']}")
    print(f"   ⭐ {repo['stars']} | {repo['today_stars']}")
    print(f"   {repo['description']}")
    print()

# 儲存為 JSON
with open("trending.json", "w", encoding="utf-8") as f:
    json.dump(repos, f, ensure_ascii=False, indent=2)
```
