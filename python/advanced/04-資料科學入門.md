### 建立陣列

```python
import numpy as np

# 從列表建立
arr = np.array([1, 2, 3, 4, 5])
print(arr)          # [1 2 3 4 5]
print(type(arr))    # <class 'numpy.ndarray'>

# 二維陣列
matrix = np.array([[1, 2, 3], [4, 5, 6]])
print(matrix)
# [[1 2 3]
#  [4 5 6]]

# 特殊陣列
zeros = np.zeros((3, 4))      # 全 0
ones = np.ones((2, 3))        # 全 1
identity = np.eye(3)          # 單位矩陣
random_arr = np.random.rand(5)  # 隨機數

# 範圍陣列
range_arr = np.arange(0, 10, 2)  # [0 2 4 6 8]
linspace = np.linspace(0, 1, 5)  # [0.   0.25 0.5  0.75 1. ]
```

### 陣列運算

```python
import numpy as np

arr = np.array([1, 2, 3, 4, 5])

# 基本運算（向量化）
print(arr + 10)        # [11 12 13 14 15]
print(arr * 2)         # [ 2  4  6  8 10]
print(arr ** 2)        # [ 1  4  9 16 25]
print(np.sqrt(arr))    # [1. 1.41 1.73 2. 2.24]

# 統計運算
print(arr.sum())       # 15
print(arr.mean())      # 3.0
print(arr.max())       # 5
print(arr.min())       # 1
print(arr.std())       # 標準差

# 條件篩選
print(arr[arr > 3])    # [4 5]
print(arr[(arr > 2) & (arr < 5)])  # [3 4]
```

### Series：一維資料

```python
import pandas as pd

# 從列表建立
s = pd.Series([10, 20, 30, 40, 50])
print(s)
# 0    10
# 1    20
# 2    30
# 3    40
# 4    50
# dtype: int64

# 自訂索引
s = pd.Series(
    [85, 92, 78, 95],
    index=["Alice", "Bob", "Charlie", "Diana"]
)
print(s)
# Alice      85
# Bob        92
# Charlie    78
# Diana      95
# dtype: int64

# 存取
print(s["Alice"])    # 85
print(s[s > 85])     # Bob 92, Diana 95
```

### DataFrame：二維資料

```python
import pandas as pd

# 從字典建立
data = {
    "name": ["Alice", "Bob", "Charlie", "Diana"],
    "age": [20, 22, 21, 23],
    "score": [85, 92, 78, 95],
    "city": ["台北", "台中", "高雄", "台北"]
}

df = pd.DataFrame(data)
print(df)
#       name  age  score city
# 0    Alice   20     85  台北
# 1      Bob   22     92  台中
# 2  Charlie   21     78  高雄
# 3    Diana   23     95  台北

# 基本資訊
print(df.info())
print(df.describe())
print(df.shape)      # (4, 4)
```

### 資料選取

```python
# 選取欄位
print(df["name"])                # 單一欄位（Series）
print(df[["name", "score"]])     # 多個欄位（DataFrame）

# 選取列（loc / iloc）
print(df.loc[0])                 # 第一列（標籤）
print(df.iloc[0])                # 第一列（整數位置）
print(df.loc[1:3])               # 第 2-4 列
print(df.iloc[1:3])              # 第 2-3 列

# 條件篩選
print(df[df["score"] >= 90])     # 成績 >= 90
print(df[(df["score"] >= 80) & (df["city"] == "台北")])
```

### 資料清理

```python
import pandas as pd
import numpy as np

# 建立包含缺失值的資料
data = {
    "name": ["Alice", "Bob", None, "Diana"],
    "age": [20, None, 21, 23],
    "score": [85, 92, None, 95]
}
df = pd.DataFrame(data)

# 檢查缺失值
print(df.isnull())
print(df.isnull().sum())

# 處理缺失值
df_drop = df.dropna()                           # 移除有缺失值的列
df_fill = df.fillna(0)                          # 填補為 0
df["age"] = df["age"].fillna(df["age"].mean())  # 填補平均值
df["name"] = df["name"].fillna("未知")          # 填補文字

# 移除重複
df = df.drop_duplicates()
```

### 資料分組與聚合

```python
import pandas as pd

# 範例資料
data = {
    "city": ["台北", "台中", "高雄", "台北", "台中"],
    "product": ["A", "B", "A", "B", "A"],
    "sales": [100, 150, 120, 130, 110],
    "quantity": [10, 15, 12, 13, 11]
}
df = pd.DataFrame(data)

# 分組統計
city_summary = df.groupby("city")["sales"].agg(["sum", "mean", "count"])
print(city_summary)
#       sum   mean  count
# city
# 高雄  120  120.0      1
# 台北  230  115.0      2
# 台中  260  130.0      2

# 多欄位分組
product_summary = df.groupby(["city", "product"]).agg({
    "sales": "sum",
    "quantity": "mean"
}).round(1)
print(product_summary)
```

## 實際案例：分析 CSV 資料

```python
import pandas as pd
import matplotlib.pyplot as plt

# 讀取 CSV
df = pd.read_csv("sales_data.csv", encoding="utf-8")

# 資料概覽
print("=== 資料概覽 ===")
print(df.head())
print("\n=== 基本統計 ===")
print(df.describe())

# 資料清理
df = df.dropna()                    # 移除缺失值
df["date"] = pd.to_datetime(df["date"])  # 轉換日期格式

# 分析
# 1. 每月總銷售額
monthly_sales = df.groupby(df["date"].dt.month)["amount"].sum()

# 2. 各產品銷售排行
product_ranking = df.groupby("product")["quantity"].sum().sort_values(ascending=False)

# 3. 找出最佳客戶
top_customers = df.groupby("customer_id")["amount"].sum().nlargest(10)

# 視覺化
plt.figure(figsize=(12, 5))

plt.subplot(1, 2, 1)
monthly_sales.plot(kind="line", marker="o", title="每月銷售趨勢")
plt.xlabel("月份")
plt.ylabel("銷售額")

plt.subplot(1, 2, 2)
product_ranking.head(10).plot(kind="barh", title="熱銷產品 Top 10")
plt.xlabel("銷售數量")

plt.tight_layout()
plt.show()

# 輸出報表
report = {
    "總銷售額": df["amount"].sum(),
    "平均訂單金額": df["amount"].mean(),
    "總訂單數": len(df),
    "活躍客戶數": df["customer_id"].nunique(),
    "產品種類": df["product"].nunique()
}

print("\n=== 銷售報表 ===")
for key, value in report.items():
    print(f"{key}: {value:,.0f}")
```
