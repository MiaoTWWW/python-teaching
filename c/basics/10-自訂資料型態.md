# 10 - 自訂資料型態

> 🎯 **學習目標**：學會使用 struct、union、enum、typedef 來建立自訂資料型態，讓程式碼更具表達力與組織性。

---

## 前言

C 語言的基本型別（int、float、char...）雖然實用，但在描述真實世界的複雜資料時往往不夠用。例如一個「學生」有姓名、年齡、成績等多種屬性，我們需要一個方式將這些相關的資料打包在一起——這就是自訂資料型態的用途。

---

## 結構（struct）

`struct` 讓我們將不同型別的變數結合成一個新的資料型態。

### 宣告與定義

```c
// 定義一個「學生」結構
struct Student {
    char name[50];     // 姓名
    int age;           // 年齡
    double score;      // 成績
    char grade;        // 等第（A/B/C/D/F）
};  // ← 注意這裡有分號！
```

### 使用結構

```c
#include <stdio.h>
#include <string.h>

struct Student {
    char name[50];
    int age;
    double score;
    char grade;
};

int main() {
    // 方式一：宣告後逐個設定
    struct Student s1;
    strcpy(s1.name, "小明");   // 字串不能直接指定，要用 strcpy
    s1.age = 20;
    s1.score = 85.5;
    s1.grade = 'A';

    // 方式二：宣告時初始化（順序初始化）
    struct Student s2 = {"小華", 21, 92.0, 'A'};

    // 方式三：指定初始化（designated initializer，C99）
    struct Student s3 = {
        .name = "小美",
        .age = 19,
        .score = 78.5,
        .grade = 'B'
    };

    // 存取成員使用 . 運算子
    printf("=== 學生資料 ===\n");
    printf("姓名：%s\n", s1.name);
    printf("年齡：%d\n", s1.age);
    printf("成績：%.1f\n", s1.score);
    printf("等第：%c\n", s1.grade);

    // 結構可以整個指定
    struct Student s4 = s1;     // 複製整個結構
    printf("\n副本：%s，%d 歲\n", s4.name, s4.age);

    return 0;
}
```

### 巢狀結構

結構中可以包含其他結構：

```c
#include <stdio.h>
#include <string.h>

struct Date {
    int year;
    int month;
    int day;
};

struct Employee {
    char name[50];
    int id;
    struct Date hire_date;  // 巢狀結構
    double salary;
};

int main() {
    struct Employee e1 = {
        .name = "王經理",
        .id = 1001,
        .hire_date = {2020, 3, 15},  // 巢狀初始化
        .salary = 65000.0
    };

    // 存取巢狀成員
    printf("%s 於 %d 年 %d 月 %d 日入職\n",
           e1.name,
           e1.hire_date.year,
           e1.hire_date.month,
           e1.hire_date.day);

    return 0;
}
```

## 結構與記憶體（Padding / Alignment）

結構的大小不一定是成員大小的總和——編譯器為了效能，會在成員之間插入**填充位元組（padding）**：

```c
#include <stdio.h>

// 直覺認為大小是 1 + 4 + 8 = 13 bytes
struct Packed {
    char c;     // 1 byte
    int i;      // 4 bytes
    double d;   // 8 bytes
};

// 但實際上因為對齊（alignment）要求：
struct Packed2 {
    char c;     // 1 byte
    // 3 bytes padding ← 為了讓 i 對齊在 4 的倍數的位址
    int i;      // 4 bytes
    double d;   // 8 bytes
};

int main() {
    printf("sizeof(struct Packed)  = %zu\n", sizeof(struct Packed));
    // 很可能是 16（1 + 3 + 4 + 8），不是 13！

    // 改變成員順序可以減少 padding
    struct Optimized {
        double d;   // 8 bytes
        int i;      // 4 bytes
        char c;     // 1 byte
        // 3 bytes padding ← 為了整體結構對齊
    };
    printf("sizeof(struct Optimized) = %zu\n", sizeof(struct Optimized));
    // 很可能是 16，還是 16

    return 0;
}
```

### 對齊規則

1. 每個型別有各自的**對齊要求**（alignment requirement）：
   - `char`：1 byte 對齊（任何位址都可以）
   - `short`：2 bytes 對齊
   - `int`：4 bytes 對齊
   - `double`：8 bytes 對齊（在 64 位元系統上）
2. 結構成員會按照對齊要求，被放在適當的位址上
3. 結構的總大小是最大成員對齊要求的倍數

> 💡 **提示**：將結構成員**從大到小排列**（double → int → char）通常可以減少 padding，節省記憶體。但不需過度最佳化——除非你要建立上千萬個結構實例。

### 使用 offsetof 觀察 padding

```c
#include <stdio.h>
#include <stddef.h>  // offsetof

struct Example {
    char a;
    int b;
    char c;
};

int main() {
    printf("offsetof(a) = %zu\n", offsetof(struct Example, a));  // 0
    printf("offsetof(b) = %zu\n", offsetof(struct Example, b));  // 4（不是 1！有 3 bytes padding）
    printf("offsetof(c) = %zu\n", offsetof(struct Example, c));  // 8
    printf("sizeof     = %zu\n", sizeof(struct Example));        // 12（不是 6！）

    return 0;
}
```

## `->` 運算子：指向結構的指標

當我們用指標指向結構時，需要用 `->` 運算子來存取成員：

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

struct Student {
    char name[50];
    int age;
};

int main() {
    struct Student s = {"小明", 20};
    struct Student *p = &s;

    // 使用 . 運算子（透過結構本身）
    printf("姓名：%s\n", s.name);

    // 使用 -> 運算子（透過結構指標）
    printf("年齡：%d\n", p->age);

    // 以下兩種寫法等價：
    p->age = 21;          // 方式一：-> 運算子
    (*p).age = 21;        // 方式二：解參考後再用 . （比較麻煩）

    // 動態配置的結構
    struct Student *ps = (struct Student*)malloc(sizeof(struct Student));
    if (ps != NULL) {
        strcpy(ps->name, "小華");  // 用 -> 存取
        ps->age = 22;
        printf("%s 是 %d 歲\n", ps->name, ps->age);
        free(ps);
    }

    return 0;
}
```

> 💡 **提示**：`p->member` 是 `(*p).member` 的語法糖。在操作動態配置的結構（經由指標存取）時，`->` 幾乎總是使用。

### 傳遞結構給函式

```c
#include <stdio.h>
#include <string.h>

struct Point {
    int x;
    int y;
};

// ❌ 傳值：複製整個結構（當結構很大時效能差）
void print_point(struct Point p) {
    printf("(%d, %d)\n", p.x, p.y);
}

// ✅ 傳址：只複製指標（較快，且可以修改原結構）
void move_point(struct Point *p, int dx, int dy) {
    p->x += dx;
    p->y += dy;
}

// ✅ 使用 const 保護內容（如果不修改）
void display_point(const struct Point *p) {
    printf("(%d, %d)\n", p->x, p->y);
}

int main() {
    struct Point pt = {3, 4};

    print_point(pt);       // 傳值
    move_point(&pt, 1, 2); // 傳址
    display_point(&pt);    // 傳址（唯讀）

    return 0;
}
```

### 函式回傳結構

```c
#include <stdio.h>

struct Point {
    int x;
    int y;
};

struct Point create_point(int x, int y) {
    struct Point p = {x, y};
    return p;  // 回傳整個結構（會複製）
}

int main() {
    struct Point p1 = create_point(10, 20);
    printf("(%d, %d)\n", p1.x, p1.y);
    return 0;
}
```

> ⚠️ **注意**：回傳大型結構時會有複製的開銷。如果結構很大，考慮回傳指標（但要確保指向的記憶體在函式結束後仍然有效）。

## typedef — 型別別名

`typedef` 可以為既有的型別建立新的名稱，讓程式碼更簡潔：

```c
typedef unsigned char byte;     // byte 就是 unsigned char
typedef int* int_ptr;           // int_ptr 就是 int*
```

### 對 struct 使用 typedef

這是最常見的用法——省去每次都要寫 `struct` 關鍵字的麻煩：

```c
// 不使用 typedef
struct Student {
    char name[50];
    int age;
};
struct Student s1;    // 每次都要寫 struct

// 使用 typedef（方式一：在定義時加上）
typedef struct {
    char name[50];
    int age;
} Student;
Student s2;           // 不需要 struct 關鍵字

// 使用 typedef（方式二：先定義再取別名）
struct Point {
    int x;
    int y;
};
typedef struct Point Point;
Point p1;
struct Point p2;      // 兩種寫法都可以
```

```c
#include <stdio.h>

typedef struct {
    char title[100];
    char author[50];
    int year;
    double price;
} Book;

int main() {
    Book b1 = {"C 語言程式設計", "Dennis Ritchie", 1978, 39.99};
    Book b2 = {"C 陷阱與缺陷", "Andrew Koenig", 1988, 29.99};

    printf("書名：%s\n", b1.title);
    printf("作者：%s\n", b1.author);
    printf("出版年份：%d\n", b1.year);

    return 0;
}
```

### typedef vs #define

| | typedef | #define |
|--|---------|--------|
| 處理時機 | 編譯時期 | 前置處理（文字取代） |
| 型別檢查 | ✅ 有 | ❌ 無 |
| 作用域 | 區塊作用域 | 檔案作用域（直到 #undef） |
| 可以搭配指標 | ✅ `typedef int* int_p;` | ⚠️ `#define int_p int*` 有陷阱 |

```c
#define PINT int*
typedef int* int_ptr;

PINT a, b;     // a 是 int*，但 b 只是 int ❌
int_ptr c, d;  // c 和 d 都是 int* ✅
```

## 聯合（union）

`union` 與 `struct` 類似，但所有成員**共用同一塊記憶體**：

```c
union Data {
    int i;
    float f;
    char c;
};
```

```c
#include <stdio.h>

union Data {
    int i;
    float f;
    char c;
};

int main() {
    union Data d;

    printf("sizeof(union Data) = %zu\n", sizeof(union Data));  // 4（最大成員的大小）

    d.i = 42;
    printf("作為整數：%d\n", d.i);

    d.f = 3.14f;      // 設定 f 時，i 的內容就被覆蓋了
    printf("作為浮點數：%f\n", d.f);
    printf("作為整數（已損毀）：%d\n", d.i);  // ⚠️ 垃圾值

    return 0;
}
```

### 記憶體示意圖

```
union Data:
  位址： 0x1000  0x1001  0x1002  0x1003
        ┌──────┬──────┬──────┬──────┐
        │            │            │
        └──────┴──────┴──────┴──────┘
          ↑ int i          ↑ float f
                              ↑ char c
  所有成員從同一個起始位址開始
```

### union 的應用：檢查位元組序（Endianness）

```c
#include <stdio.h>

union EndianCheck {
    int i;
    unsigned char bytes[4];
};

int main() {
    union EndianCheck e;
    e.i = 0x12345678;

    printf("整數：0x%X\n", e.i);
    printf("位元組：");
    for (int i = 0; i < 4; i++) {
        printf("%02X ", e.bytes[i]);
    }
    printf("\n");

    if (e.bytes[0] == 0x78) {
        printf("→ 小端序（Little Endian）\n");  // 常見於 x86/x64
    } else {
        printf("→ 大端序（Big Endian）\n");     // 常見於網路協定
    }

    return 0;
}
```

> 💡 **提示**：union 常用於：節省記憶體（不同時間使用不同型別）、資料型別轉換（reinterpretation）、通訊協定解析（同一組位元有多種解讀方式）。

## 列舉（enum）

`enum` 定義一組具名的整數常數，讓程式碼更具可讀性：

```c
#include <stdio.h>

// 定義一週的天數
enum Weekday {
    MON,     // 預設為 0
    TUE,     // 1
    WED,     // 2
    THU,     // 3
    FRI,     // 4
    SAT,     // 5
    SUN      // 6
};

int main() {
    enum Weekday today = WED;

    printf("今天是週三（列舉值：%d）\n", today);

    // switch 搭配 enum 非常好用
    switch (today) {
        case MON: printf("星期一，上班日\n"); break;
        case TUE: printf("星期二\n"); break;
        case WED: printf("星期三，小週末\n"); break;
        case THU: printf("星期四\n"); break;
        case FRI: printf("星期五，TGIF！\n"); break;
        case SAT:
        case SUN: printf("週末！\n"); break;
    }

    return 0;
}
```

### 自訂列舉值

```c
#include <stdio.h>

// 自訂起始值
enum Color {
    RED = 1,
    GREEN = 3,
    BLUE = 5
};

// 位元旗標常數
enum Permission {
    READ    = 1 << 0,  // 1
    WRITE   = 1 << 1,  // 2
    EXEC    = 1 << 2,  // 4
    ADMIN   = 1 << 3   // 8
};

// 自動遞增
enum ErrorCode {
    SUCCESS = 0,
    FILE_NOT_FOUND,    // 1
    PERMISSION_DENIED, // 2
    INVALID_INPUT,     // 3
    OUT_OF_MEMORY      // 4
};

int main() {
    printf("RED = %d, GREEN = %d, BLUE = %d\n", RED, GREEN, BLUE);
    printf("READ = %d, WRITE = %d, EXEC = %d, ADMIN = %d\n",
           READ, WRITE, EXEC, ADMIN);
    printf("SUCCESS = %d, FILE_NOT_FOUND = %d\n", SUCCESS, FILE_NOT_FOUND);

    // enum 提升程式碼可讀性的經典範例
    enum Permission perms = READ | WRITE;
    if (perms & READ) printf("有讀取權限\n");

    return 0;
}
```

### enum vs #define

```c
// 使用 #define
#define MON 0
#define TUE 1
#define WED 2
// 沒有型別檢查，純文字取代

// 使用 enum
enum Day { MON, TUE, WED };
// 有型別檢查，除錯器可以顯示名稱
// 自動編號，不需要手動給值
```

> 💡 **提示**：在現代 C 程式設計中，偏好使用 `enum` 而非 `#define` 來定義相關的整數常數，因為它有型別檢查且對除錯工具更友善。

## 位元欄位（Bit Field）

位元欄位讓我們可以精確控制結構成員佔用的**位元數**，常用於嵌入式系統和通訊協定：

```c
#include <stdio.h>

// 使用位元欄位節省記憶體
struct Flags {
    unsigned int is_ready   : 1;   // 只用 1 個位元（0 或 1）
    unsigned int is_error   : 1;   // 只用 1 個位元
    unsigned int mode       : 2;   // 只用 2 個位元（0-3）
    unsigned int error_code : 4;   // 只用 4 個位元（0-15）
};

int main() {
    struct Flags f = {0};

    f.is_ready = 1;
    f.is_error = 0;
    f.mode = 2;
    f.error_code = 7;

    printf("is_ready: %u\n", f.is_ready);
    printf("mode: %u\n", f.mode);
    printf("error_code: %u\n", f.error_code);
    printf("sizeof(struct Flags) = %zu byte\n", sizeof(struct Flags));
    // 只有 4 bytes，但如果你用 4 個 unsigned int，會是 16 bytes

    return 0;
}
```

> 💡 **提示**：位元欄位的位元數不能超過基礎型別的位元數（例如 `unsigned int` 最多 32 位元）。不同編譯器對位元欄位的記憶體佈局（bit-endianness）可能有差異，跨平台時要注意。

## 本章重點整理

| 概念 | 重點 |
|------|------|
| `struct` | 將不同型別的資料組合成新型態 |
| `.` 運算子 | 存取結構成員 |
| `->` 運算子 | 透過指標存取結構成員 |
| Padding | 編譯器為了對齊在成員之間插入的填充位元組 |
| `typedef` | 為型別建立別名（常用來簡化 struct 寫法） |
| `union` | 所有成員共用同一塊記憶體 |
| `enum` | 定義一組具名的整數常數 |
| 位元欄位 | 精確控制成員的位元數（`: N` 語法） |
