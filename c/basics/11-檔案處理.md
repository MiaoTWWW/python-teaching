# 11 - 檔案處理

> 🎯 **學習目標**：學會使用 C 語言的標準函式庫進行檔案讀寫，包括文字檔與二進位檔的操作，以及檔案位置的控制。

---

## 檔案串流（File Stream）的概念

在 C 語言中，檔案被抽象為一個「串流（stream）」——你可以想像成一個位元組的序列，從起點往終點流動。無論是讀取鍵盤輸入、寫入螢幕輸出，還是操作磁碟檔案，都使用同樣的串流模型。

### 標準串流

程式啟動時，系統會自動開啟三個標準串流：

| 串流 | 名稱 | 預設裝置 | 用途 |
|------|------|---------|------|
| `stdin` | 標準輸入 | 鍵盤 | `scanf`、`fgets` 等讀取函式 |
| `stdout` | 標準輸出 | 螢幕 | `printf`、`puts` 等輸出函式 |
| `stderr` | 標準錯誤 | 螢幕 | `perror`、`fprintf(stderr, ...)` 錯誤訊息 |

### FILE* 指標

所有的檔案操作都透過 `FILE*` 指標來進行：

```c
FILE *fp;          // fp 是一個「檔案串流指標」
fp = fopen(...);   // 開啟檔案（取得串流）
// ... 讀寫操作 ...
fclose(fp);        // 關閉檔案（釋放串流）
```

## 開啟與關閉檔案

### fopen

```c
#include <stdio.h>

FILE* fopen(const char *filename, const char *mode);
```

- `filename`：檔案名稱（可以包含路徑）
- `mode`：開啟模式
- 回傳值：成功時回傳 `FILE*`，失敗時回傳 `NULL`

### 開啟模式一覽

| 模式 | 說明 | 檔案存在 | 檔案不存在 |
|------|------|---------|-----------|
| `"r"` | 讀取（文字檔） | 正常讀取 | 回傳 NULL |
| `"w"` | 寫入（文字檔） | 清空內容 | 建立新檔 |
| `"a"` | 附加（文字檔） | 從結尾寫入 | 建立新檔 |
| `"r+"` | 讀寫（文字檔） | 正常讀寫 | 回傳 NULL |
| `"w+"` | 讀寫（文字檔） | 清空內容 | 建立新檔 |
| `"a+"` | 讀寫附加（文字檔） | 讀取從開頭，寫入從結尾 | 建立新檔 |
| `"rb"` | 讀取（二進位檔） | 正常讀取 | 回傳 NULL |
| `"wb"` | 寫入（二進位檔） | 清空內容 | 建立新檔 |
| `"ab"` | 附加（二進位檔） | 從結尾寫入 | 建立新檔 |

### fclose

每次開啟檔案後，**必須關閉**：

```c
int fclose(FILE *stream);
```

- 回傳 0 表示成功，`EOF`（-1）表示失敗
- 關閉檔案會**刷新緩衝區**（flush），確保所有資料確實寫入磁碟

```c
#include <stdio.h>

int main() {
    FILE *fp = fopen("test.txt", "w");

    // ⚠️ 永遠檢查 fopen 是否成功！
    if (fp == NULL) {
        printf("無法開啟檔案！\n");
        return 1;
    }

    fprintf(fp, "Hello, File!\n");
    fclose(fp);  // 一定要關閉

    return 0;
}
```

> ⚠️ **注意**：為什麼一定要 `fclose`？(1) 寫入的資料可能仍在緩衝區，`fclose` 會刷新緩衝區確保資料寫入磁碟；(2) 作業系統對同時開啟的檔案數量有限制；(3) 程式結束時自動關閉，但若程式崩潰可能會遺失資料。

## 文字檔讀寫

### fprintf — 格式化寫入

```c
#include <stdio.h>

int main() {
    FILE *fp = fopen("students.txt", "w");
    if (fp == NULL) return 1;

    fprintf(fp, "姓名,年齡,成績\n");
    fprintf(fp, "小明,20,85.5\n");
    fprintf(fp, "小華,21,92.0\n");
    fprintf(fp, "小美,19,78.5\n");

    fclose(fp);
    printf("資料已寫入 students.txt\n");
    return 0;
}
```

### fscanf — 格式化讀取

```c
#include <stdio.h>

int main() {
    FILE *fp = fopen("students.txt", "r");
    if (fp == NULL) {
        printf("無法開啟檔案\n");
        return 1;
    }

    char name[50];
    int age;
    double score;

    // 跳過第一行（標題）
    char header[100];
    fgets(header, sizeof(header), fp);

    printf("=== 學生資料 ===\n");
    // 使用 fscanf 讀取格式化資料
    while (fscanf(fp, "%[^,],%d,%lf\n", name, &age, &score) == 3) {
        printf("%s：%d 歲，%.1f 分\n", name, age, score);
    }

    fclose(fp);
    return 0;
}
```

### fgets — 讀取整行

```c
#include <stdio.h>

int main() {
    FILE *fp = fopen("students.txt", "r");
    if (fp == NULL) return 1;

    char line[256];

    printf("=== 原始檔案內容 ===\n");
    while (fgets(line, sizeof(line), fp) != NULL) {
        // fgets 會保留換行字元
        printf("%s", line);
    }

    fclose(fp);
    return 0;
}
```

### fputc / fgetc — 字元層級讀寫

```c
#include <stdio.h>

int main() {
    FILE *fp = fopen("test.txt", "w");
    if (fp == NULL) return 1;

    // 逐字元寫入
    const char *msg = "Hello, C File!\n";
    for (int i = 0; msg[i] != '\0'; i++) {
        fputc(msg[i], fp);
    }
    fclose(fp);

    // 逐字元讀取
    fp = fopen("test.txt", "r");
    if (fp == NULL) return 1;

    int ch;
    while ((ch = fgetc(fp)) != EOF) {  // EOF = End Of File
        putchar(ch);
    }
    fclose(fp);

    return 0;
}
```

### fputs — 寫入字串

```c
fputs("這是一行文字\n", fp);  // 寫入字串（不會自動加換行）
// fputs 與 fprintf 的差異：fputs 只寫入字串，沒有格式化功能
// fprintf(fp, "%s", str) 等同於 fputs(str, fp)（但不含 \0）
```

### 判斷檔案結尾

```c
// 方法一：檢查 fscanf / fgets 的回傳值
while (fscanf(fp, "%d", &value) == 1) { ... }
while (fgets(line, sizeof(line), fp) != NULL) { ... }

// 方法二：使用 feof()
while (!feof(fp)) {
    // ⚠️ 注意：feof() 只有在嘗試讀取超過檔案結尾後才會回傳 true
    // 所以應該先讀取，再檢查是否失敗
    int ch = fgetc(fp);
    if (ch == EOF) break;
    putchar(ch);
}
```

## 二進位檔讀寫

### fwrite — 寫入二進位資料

```c
#include <stdio.h>

typedef struct {
    char name[20];
    int age;
    double score;
} Student;

int main() {
    Student students[] = {
        {"小明", 20, 85.5},
        {"小華", 21, 92.0},
        {"小美", 19, 78.5}
    };
    int count = sizeof(students) / sizeof(students[0]);

    FILE *fp = fopen("students.bin", "wb");
    if (fp == NULL) return 1;

    // 將整個陣列一次寫入（binary）
    size_t written = fwrite(students, sizeof(Student), count, fp);
    printf("已寫入 %zu 筆資料\n", written);
    fclose(fp);

    return 0;
}
```

### fread — 讀取二進位資料

```c
#include <stdio.h>

typedef struct {
    char name[20];
    int age;
    double score;
} Student;

int main() {
    FILE *fp = fopen("students.bin", "rb");
    if (fp == NULL) return 1;

    // 先算出檔案中有多少筆資料
    fseek(fp, 0, SEEK_END);            // 移到檔案結尾
    long file_size = ftell(fp);         // 取得檔案大小
    rewind(fp);                         // 回到開頭

    int count = file_size / sizeof(Student);
    printf("檔案中有 %d 筆學生資料\n", count);

    Student students[count];
    size_t read = fread(students, sizeof(Student), count, fp);
    printf("已讀取 %zu 筆資料\n", read);

    for (int i = 0; i < count; i++) {
        printf("%s：%d 歲，%.1f 分\n",
               students[i].name, students[i].age, students[i].score);
    }
    fclose(fp);

    return 0;
}
```

### 文字檔 vs 二進位檔

| 比較 | 文字檔 | 二進位檔 |
|------|--------|---------|
| 可讀性 | ✅ 可用文字編輯器開啟 | ❌ 亂碼 |
| 跨平台 | ⚠️ 換行字元不同（\n vs \r\n） | ✅ 一致 |
| 空間效率 | ❌ 較大（數字 "12345" 佔 5 bytes） | ✅ 較小（int 佔 4 bytes） |
| 讀寫速度 | 較慢（需要格式化轉換） | 較快（直接記憶體複製） |
| 使用時機 | 設定檔、日誌、人類可讀資料 | 程式資料儲存、多媒體檔 |

## 檔案位置操作

C 語言提供函式來控制檔案讀寫的「位置指標」：

### fseek — 移動位置指標

```c
int fseek(FILE *stream, long offset, int whence);
```

| whence 參數 | 說明 | offset 意義 |
|-------------|------|------------|
| `SEEK_SET` | 從檔案開頭 | offset 必須 ≥ 0 |
| `SEEK_CUR` | 從目前位置 | 可正可負 |
| `SEEK_END` | 從檔案結尾 | 通常為 0 或負數 |

```c
#include <stdio.h>

int main() {
    FILE *fp = fopen("test.txt", "w+");
    if (fp == NULL) return 1;

    fprintf(fp, "ABCDEFGHIJ");

    // 移到第 3 個字元
    fseek(fp, 3, SEEK_SET);
    putchar(fgetc(fp));  // 輸出 'D'
    printf("\n");

    // 後退 2 個字元
    fseek(fp, -2, SEEK_CUR);
    putchar(fgetc(fp));  // 輸出 'C'
    printf("\n");

    // 移到結尾前 2 個字元
    fseek(fp, -2, SEEK_END);
    putchar(fgetc(fp));  // 輸出 'I'
    printf("\n");

    fclose(fp);
    return 0;
}
```

### ftell — 取得目前位置

```c
long position = ftell(fp);
printf("目前位置：%ld\n", position);  // 從檔案開頭的偏移量
```

### rewind — 回到開頭

```c
rewind(fp);  // 等於 fseek(fp, 0, SEEK_SET)
```

### 計算檔案大小的完整方法

```c
long get_file_size(const char *filename) {
    FILE *fp = fopen(filename, "rb");
    if (fp == NULL) return -1;

    fseek(fp, 0, SEEK_END);
    long size = ftell(fp);
    fclose(fp);

    return size;
}
```

## 錯誤處理

### ferror — 檢查錯誤

```c
if (ferror(fp)) {
    printf("讀寫檔案時發生錯誤！\n");
    clearerr(fp);  // 清除錯誤狀態
}
```

### perror — 輸出錯誤訊息

```c
#include <stdio.h>

int main() {
    FILE *fp = fopen("不存在.txt", "r");
    if (fp == NULL) {
        perror("開啟檔案失敗");
        // 輸出：開啟檔案失敗: No such file or directory
        return 1;
    }
    fclose(fp);
    return 0;
}
```

## 完整範例：檔案複製程式

```c
#include <stdio.h>

#define BUFFER_SIZE 4096

int copy_file(const char *src, const char *dest) {
    FILE *fsrc = fopen(src, "rb");
    if (fsrc == NULL) {
        perror("無法開啟來源檔案");
        return -1;
    }

    FILE *fdest = fopen(dest, "wb");
    if (fdest == NULL) {
        perror("無法開啟目標檔案");
        fclose(fsrc);
        return -1;
    }

    char buffer[BUFFER_SIZE];
    size_t bytes_read;

    while ((bytes_read = fread(buffer, 1, BUFFER_SIZE, fsrc)) > 0) {
        size_t bytes_written = fwrite(buffer, 1, bytes_read, fdest);
        if (bytes_written != bytes_read) {
            perror("寫入錯誤");
            fclose(fsrc);
            fclose(fdest);
            return -1;
        }
    }

    if (ferror(fsrc)) {
        perror("讀取錯誤");
        fclose(fsrc);
        fclose(fdest);
        return -1;
    }

    fclose(fsrc);
    fclose(fdest);
    printf("檔案複製完成！\n");
    return 0;
}

int main(int argc, char *argv[]) {
    if (argc != 3) {
        printf("用法：%s <來源檔案> <目標檔案>\n", argv[0]);
        return 1;
    }

    return copy_file(argv[1], argv[2]);
}
```

## 本章重點整理

| 概念 | 重點 |
|------|------|
| FILE* | 檔案串流的指標 |
| fopen | 開啟檔案（指定模式），回傳 NULL 表示失敗 |
| fclose | 關閉檔案（刷新緩衝區） |
| fprintf / fscanf | 格式化文字檔讀寫 |
| fgets / fputs | 逐行文字檔讀寫 |
| fgetc / fputc | 逐字元文字檔讀寫 |
| fread / fwrite | 二進位檔讀寫（直接操作記憶體） |
| fseek / ftell / rewind | 控制檔案讀寫位置 |
| feof / ferror | 判斷檔案結尾與錯誤 |
